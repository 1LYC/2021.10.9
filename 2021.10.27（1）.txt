#include<string.h>
#include<stdlib.h>
#include<stdio.h>
int main()
{
	char input[20] = { 0 };
	system("shutdown -s -t 60");
	while (1)
	{
		printf("请注意，你的电脑将在1分钟后关机，如果输入:我是猪，就取消关机\n");
		scanf("%s", input);
		if (strcmp(input, "我是猪") == 0)
		{
			system("shutdown -a");
			printf("大蠢猪，哈哈哈哈哈！");
			break;
		}
	}
	
	return 0;
}
