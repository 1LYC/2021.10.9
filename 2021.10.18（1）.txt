#include<stdio.h>
//写一个代码：打印100到200之间的素数
//素数素数只能被1和他本身整除的数字
int main()
{
	int i = 0;
	int count = 0;
	for (i = 100;i <= 200;i++)
	{
		//判断i是否为素数
		//可以那2->i-1整除
		int j = 0;
		for (j = 2;j < i;j++)
		{
			if (i % j == 0)
				break;
		}
		//1.2
		if (j >= i)
		{
			printf("%d\n", i);
			count++;
		}
	}
	printf("\ncount=%d\n", count);
	return 0;
}
