#include<stdio.h>                                      /*包含头文件*/
#define HEG 10                                          /*定义常量*/
int calculate(int Long, int Width);                      /*函数声明*/
int main()                                              /*主函数main*/
{
	int m_Long;                                          /*定义整型函数，表示长度*/
	int m_Width;                                         /*定义整型函数，表示宽度*/
	int result;                                          /*定义整型函数，表示长方体的体积*/

	printf("长方的高度为:%d\n", HEG);                     /*显示提示*/

	printf("请输入长度\n");                              /*显示提示*/
	scanf_s("%d", &m_Long);                             /*输入长方的长度*/

	printf("请输入宽度\n");                              /*显示提示*/
	scanf_s("%d", &m_Width);                             /*输入长方体的宽度*/

	result = calculate(m_Long, m_Width);                     /*调动函数,计算体积*/
	printf("长方体的体积是: ");                           /*显示提示*/
	printf("%d\n", result);                                /*输出体积大小*/
	return 0;                                             /*返回整形0*/
}


int calculate(int Long, int Width)                        /*定义计算体积函数*/
{
	int result = Long * Width * HEG;                            /*具体计算体积*/
	return result;                                        /*将计算体积结果返回*/
}
