#include<stdio.h>
int main()
{
	int iNumber1, iNumber2, iNumber3;                                                 /*声明函数*/
	iNumber1 = 3;                                                                     /*为变量赋值*/
	iNumber2 = 7;


	printf("the first number is :%d\d", iNumber1);                                    /*显示变量值*/
	printf("the second number is :%d", iNumber2);

	iNumber3 = iNumber1 + 10;                                                         /*表达式中利用iNumber1变量加上一个常量*/
	printf("the first number add 10 is :%d\n", iNumber3);                              /*显示变量iNumber3的值*/

	iNumber3 = iNumber2 + 10;                                                          /*表达式中利用iNumber2变量加上一个常量*/
	printf("the secong number sdd 10 is :%d\n", iNumber3);                              /*显示变量iNumber3的值*/

	iNumber3 = iNumber1 + iNumber2;                                                     /*表达式中是两个变量进行计算*/
	printf("the result number of first add second is :%d\n", iNumber3);                  /*将程序结果输出*/
	return 0;                                                                           /*程序结束*/
}
