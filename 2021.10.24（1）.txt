#include<stdio.h>
#include<conio.h>
#include<windows.h>
//控制台句柄
HANDLE hOut;

/**
*获取光标位置
*/
void gotoxy(int x, int y)
{
	COORD pos;
	pos.X = x;//横坐标
	pos.Y = y;//纵坐标
	SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), pos);
}

int main()
{
	int n;
	int i, j = 1;
	for (i = 9;i <= 20;i++)//循环y纵坐标，打印输出上下边框===
	{
		for (j = 15;j <= 60;j++)//循环x横坐标，打印输出左右边框||
		{
			gotoxy(j, i);
			if (i == 9 || i == 20)printf("=");//输出上下边框
			else if (j == 15 || j == 59)printf("||");//输出左右边框
		}
	}
	gotoxy(25, 12);//设置显示位置
	printf("1.开始游戏");//输出文字"1.开始游戏
	gotoxy(40, 12);
	printf("2.按键说明");
	gotoxy(25, 17);
	printf("3.游戏规则");
	gotoxy(40, 17);
	printf("4.退出");
	gotoxy(21, 22);
	printf("请选择是[1 2 3 4 ]:[]\b\b");//输入选项
	scanf_s("%d", &n);
	switch (n)
	{
	case 1:
		printf("\n\t你选择了1.开始游戏'选项'");
		break;
	case 2:
		printf("\n\t你选择了2.按键说明'选项'");
		break;
	case 3:
		printf("\n\t你选择了3.游戏规则'选项'");
		break;
	case 4:
		printf("\n\t你选择了4.退出'选择'");
		break;
	}
}
